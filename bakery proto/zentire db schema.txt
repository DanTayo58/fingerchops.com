-- =====================================================
-- DATABASE: fingerchops_bakery
-- VERSION: 4.0 (Positions removed, role-based system)
-- =====================================================

DROP DATABASE IF EXISTS fingerchops_bakery;
CREATE DATABASE fingerchops_bakery CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE fingerchops_bakery;

-- =====================================================
-- 1. DEPARTMENTS
-- =====================================================
CREATE TABLE departments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    dept_code VARCHAR(10) UNIQUE NOT NULL,
    dept_name VARCHAR(50) NOT NULL,
    dept_head_id INT NULL,
    parent_dept_id INT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_dept_code (dept_code)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 2. BRANCHES
-- =====================================================
CREATE TABLE branches (
    id INT AUTO_INCREMENT PRIMARY KEY,
    branch_code VARCHAR(10) UNIQUE NOT NULL,
    branch_name VARCHAR(100) NOT NULL,
    branch_manager_id INT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    branch_fee DECIMAL(5,2) NOT NULL DEFAULT 1.00,
    address TEXT NULL,
    phone VARCHAR(20) NULL,
    email VARCHAR(100) NULL,
    is_hq TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_branch_code (branch_code),
    INDEX idx_is_hq (is_hq)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 3. ROLES (with department binding and occupancy limits)
-- =====================================================
CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    role_code VARCHAR(20) NOT NULL,
    role_name VARCHAR(50) NOT NULL UNIQUE,
    department_id INT NULL COMMENT 'NULL = global role, otherwise department‑specific',
    max_occupants INT DEFAULT 1 NOT NULL COMMENT '0 = unlimited',
    privilege_level INT NOT NULL,
    description TEXT NULL,
    can_manage_users BOOLEAN DEFAULT FALSE,
    can_approve_requests BOOLEAN DEFAULT FALSE,
    can_manage_budget BOOLEAN DEFAULT FALSE,
    can_view_reports BOOLEAN DEFAULT FALSE,
    can_manage_system BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_privilege_level (privilege_level),
    INDEX idx_role_code (role_code),
    INDEX idx_dept (department_id),
    FOREIGN KEY (department_id) REFERENCES departments(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 4. BAKERY_USERS (with branch)
-- =====================================================
CREATE TABLE bakery_users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(50) UNIQUE NOT NULL,
    fullname VARCHAR(100) NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    user_type ENUM('customer', 'vendor', 'staff') NOT NULL DEFAULT 'customer',
    branch_id INT NULL COMMENT 'NULL = headquarters or unassigned',
    default_branch_id INT NULL,
    phone VARCHAR(15) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    password_reset_token VARCHAR(64) NULL,
    password_reset_expires DATETIME NULL,
    last_password_change DATETIME NULL,
    force_password_change BOOLEAN DEFAULT FALSE,
    is_active BOOLEAN DEFAULT TRUE,
    is_verified BOOLEAN DEFAULT FALSE,
    failed_login_attempts INT DEFAULT 0,
    last_failed_login DATETIME NULL,
    account_locked_until DATETIME NULL,
    last_session_id VARCHAR(128) NULL,
    last_ip_address VARCHAR(45) NULL,
    last_user_agent TEXT NULL,
    vendor_status ENUM('pending', 'active', 'suspended', 'inactive') DEFAULT 'pending',
    wholesale_discount DECIMAL(5,2) DEFAULT 0.00,
    loyalty_points INT DEFAULT 0,
    employee_id VARCHAR(50) NULL,
    hire_date DATE NULL,
    last_login DATETIME NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_email (email),
    INDEX idx_phone (phone),
    INDEX idx_username (username),
    INDEX idx_user_id (user_id),
    INDEX idx_user_type (user_type),
    INDEX idx_branch (branch_id),
    INDEX idx_default_branch (default_branch_id),
    INDEX idx_vendor_status (vendor_status),
    INDEX idx_reset_token (password_reset_token),
    INDEX idx_account_locked (account_locked_until, is_active),
    FOREIGN KEY (branch_id) REFERENCES branches(id) ON DELETE SET NULL,
    FOREIGN KEY (default_branch_id) REFERENCES branches(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 5. USER_ROLES (assign users to roles, with department and branch context)
-- =====================================================
CREATE TABLE user_roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    role_id INT NOT NULL,
    department_id INT NULL COMMENT 'Only relevant if role is global; for department‑specific roles, it must match role.department_id',
    branch_id INT NULL,
    assigned_by INT NOT NULL,
    assigned_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expiry_date DATE NULL,
    is_active BOOLEAN DEFAULT TRUE,
    INDEX idx_user (user_id),
    INDEX idx_role (role_id),
    INDEX idx_department (department_id),
    INDEX idx_branch (branch_id),
    FOREIGN KEY (user_id) REFERENCES bakery_users(id) ON DELETE CASCADE,
    FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE CASCADE,
    FOREIGN KEY (department_id) REFERENCES departments(id) ON DELETE SET NULL,
    FOREIGN KEY (branch_id) REFERENCES branches(id) ON DELETE SET NULL,
    FOREIGN KEY (assigned_by) REFERENCES bakery_users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 6. ACTIVITY_LOGS
-- =====================================================
CREATE TABLE activity_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    action VARCHAR(255) NOT NULL,
    entity_type VARCHAR(50) NULL,
    entity_id INT NULL,
    old_data TEXT NULL,
    new_data TEXT NULL,
    ip_address VARCHAR(45) NULL,
    user_agent TEXT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user (user_id),
    INDEX idx_action (action(100)),
    INDEX idx_created (created_at),
    FOREIGN KEY (user_id) REFERENCES bakery_users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 7. PERMISSION_REQUESTS
-- =====================================================
CREATE TABLE permission_requests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    requester_id INT NOT NULL,
    requested_permission VARCHAR(100) NOT NULL,
    reason TEXT,
    status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
    approved_by INT NULL,
    approved_at DATETIME NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_status (status),
    FOREIGN KEY (requester_id) REFERENCES bakery_users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 8. PRODUCTS
-- =====================================================
CREATE TABLE products (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    base_price DECIMAL(10,2) NOT NULL,
    delivery_rate DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    category VARCHAR(50),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_category (category)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 9. NOTIFICATIONS
-- =====================================================
CREATE TABLE notifications (
    id INT AUTO_INCREMENT PRIMARY KEY,
    from_user_id INT NOT NULL,
    to_department_id INT NULL,
    to_user_id INT NULL,
    message TEXT NOT NULL,
    type ENUM('info', 'success', 'warning', 'approval', 'department') DEFAULT 'info',
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_recipient (to_department_id, to_user_id),
    INDEX idx_read (is_read),
    FOREIGN KEY (from_user_id) REFERENCES bakery_users(id) ON DELETE CASCADE,
    FOREIGN KEY (to_department_id) REFERENCES departments(id) ON DELETE CASCADE,
    FOREIGN KEY (to_user_id) REFERENCES bakery_users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 10. APPROVAL_REQUESTS
-- =====================================================
CREATE TABLE approval_requests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    item_id INT NOT NULL,
    item_type VARCHAR(50) NOT NULL,
    requester_id INT NOT NULL,
    department_id INT NOT NULL,
    status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
    response_notes TEXT,
    responded_by INT NULL,
    responded_at DATETIME NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_item (item_id, item_type),
    INDEX idx_status (status),
    FOREIGN KEY (requester_id) REFERENCES bakery_users(id),
    FOREIGN KEY (department_id) REFERENCES departments(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 11. LOGIN_ATTEMPTS
-- =====================================================
CREATE TABLE login_attempts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ip_address VARCHAR(45) NOT NULL,
    username VARCHAR(50) NULL,
    user_id INT NULL,
    attempted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    success BOOLEAN DEFAULT FALSE,
    user_agent TEXT NULL,
    INDEX idx_ip (ip_address),
    INDEX idx_username (username),
    INDEX idx_attempted_at (attempted_at),
    FOREIGN KEY (user_id) REFERENCES bakery_users(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 12. USER_SESSIONS
-- =====================================================
CREATE TABLE user_sessions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    session_id VARCHAR(128) NOT NULL UNIQUE,
    ip_address VARCHAR(45) NOT NULL,
    user_agent TEXT NULL,
    login_time DATETIME NOT NULL,
    last_activity DATETIME NOT NULL,
    expires_at DATETIME NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    INDEX idx_session (session_id),
    INDEX idx_user_active (user_id, is_active),
    INDEX idx_expires (expires_at),
    FOREIGN KEY (user_id) REFERENCES bakery_users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 13. AUDIT_TRAIL
-- =====================================================
CREATE TABLE audit_trail (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NULL,
    action_type ENUM('LOGIN', 'LOGOUT', 'CREATE', 'UPDATE', 'DELETE', 'VIEW', 'EXPORT', 'FAILED_LOGIN') NOT NULL,
    table_name VARCHAR(50) NULL,
    record_id INT NULL,
    old_values TEXT NULL,
    new_values TEXT NULL,

    user_agent TEXT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user (user_id),
    INDEX idx_action (action_type),
    INDEX idx_created (created_at),
    FOREIGN KEY (user_id) REFERENCES bakery_users(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================

-- =====================================================

-- =====================================================

-- =====================================================

-- =====================================================
-- 21. SALES_TARGETS

    period_type ENUM('daily', 'weekly', 'monthly', 'quarterly', 'yearly') DEFAULT 'monthly',

    period_end DATE NOT NULL,
    created_by INT NOT NULL,
    branch_id INT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

-- =====================================================

    id INT AUTO_INCREMENT PRIMARY KEY,

-- =====================================================

-- =====================================================

-- =====================================================
-- 28. ORDER_ITEMS
-- =====================================================
CREATE TABLE order_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    product_id INT NOT NULL,
    quantity INT NOT NULL,
    unit_price DECIMAL(10,2) NOT NULL,
    subtotal DECIMAL(10,2) NOT NULL,
    status ENUM('pending', 'handled', 'requested', 'transferred') DEFAULT 'pending' COMMENT 'Order item fulfillment status',
    request_to_branch_id INT NULL COMMENT 'Branch ID to request stock from',
    request_status ENUM('pending', 'approved', 'rejected', 'completed') DEFAULT NULL COMMENT 'Status of stock request',
    request_notes TEXT NULL COMMENT 'Notes for stock request',
    delivery_status ENUM('pending', 'delivered', 'not_delivered', 'returned') DEFAULT 'pending',
    delivered_at DATETIME NULL,
    returned_at DATETIME NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (order_id) REFERENCES customer_orders(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id),
    FOREIGN KEY (request_to_branch_id) REFERENCES branches(id) ON DELETE SET NULL,
    INDEX idx_order (order_id),
    INDEX idx_status (status),
    INDEX idx_request_to_branch (request_to_branch_id),
    INDEX idx_delivery_status (delivery_status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 29. BRANCH_RECEIPT_POSTS (Inter-branch stock transfers)

    to_branch_id INT NOT NULL,
    order_id INT NOT NULL,
    product_id INT NOT NULL,
    quantity INT NOT NULL,
    notes TEXT,
    status ENUM('pending', 'accepted', 'rejected', 'completed') DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_from_branch (from_branch_id),
    INDEX idx_to_branch (to_branch_id),
    INDEX idx_order (order_id),
    INDEX idx_product (product_id),
    INDEX idx_status (status),
    FOREIGN KEY (from_branch_id) REFERENCES branches(id),
    FOREIGN KEY (to_branch_id) REFERENCES branches(id),
    FOREIGN KEY (order_id) REFERENCES customer_orders(id),
    FOREIGN KEY (product_id) REFERENCES products(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- =====================================================
-- 30. ORDER_RETURNS (Returns and exchanges)
-- =====================================================
CREATE TABLE order_returns (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    return_number VARCHAR(50) NOT NULL UNIQUE,

-- =====================================================
-- 45. VIEW: department_staff
-- =====================================================
CREATE OR REPLACE VIEW department_staff AS
SELECT 
    u.id,
    u.user_id,
    u.fullname,
    u.username,
    u.email,
    u.phone,
    u.user_type,
    u.is_active,
    u.hire_date,
    d.id as department_id,
    d.dept_name as department_name,
    r.role_name,
    r.privilege_level,
    ur.assigned_date
FROM bakery_users u
JOIN user_roles ur ON u.id = ur.user_id AND ur.is_active = 1
JOIN roles r ON ur.role_id = r.id
LEFT JOIN departments d ON (r.department_id = d.id OR ur.department_id = d.id)
WHERE u.user_type = 'staff';

-- 46. VIEW: public_user_view

CREATE OR REPLACE VIEW public_user_view AS
SELECT id, user_id, username, user_type, 
       CONCAT(LEFT(phone, 4), '****', RIGHT(phone, 4)) as phone_masked,
       CONCAT(LEFT(email, 3), '***@', SUBSTRING_INDEX(email, '@', -1)) as email_masked,
       is_verified,
       created_at
FROM bakery_users;

-- =====================================================
-- 47. VIEW: products_with_stock_status
-- =====================================================
CREATE OR REPLACE VIEW products_with_stock_status AS
SELECT 
    p.id,
    p.name,
    p.description,
    p.base_price,
    p.category,
    COALESCE(SUM(ps.quantity), 0) as stock_quantity,
    CASE 
        WHEN COALESCE(SUM(ps.quantity), 0) <= 0 THEN 'Out of Stock'
        WHEN COALESCE(SUM(ps.quantity), 0) <= 5 THEN 'Critical Stock'
        WHEN COALESCE(SUM(ps.quantity), 0) <= 10 THEN 'Low Stock'
        ELSE 'In Stock'
    END as stock_status
FROM products p
LEFT JOIN product_stock ps ON p.id = ps.product_id
WHERE p.is_active = 1
GROUP BY p.id;

-- =====================================================
-- 48. VIEW: product_sales_performance
-- =====================================================
CREATE OR REPLACE VIEW product_sales_performance AS
SELECT 
    p.id as product_id,
    p.name as product_name,
    p.base_price,
    p.category,
    COALESCE(st.target_quantity, 0) as target_quantity,
    COALESCE(SUM(sr.quantity_sold), 0) as actual_sold,
    CASE 
        WHEN COALESCE(st.target_quantity, 0) > 0 
        THEN ROUND((COALESCE(SUM(sr.quantity_sold), 0) / st.target_quantity) * 100, 1)
        ELSE 0
    END as percentage_achieved,
    COALESCE(SUM(sr.quantity_sold * sr.sale_price), 0) as revenue_generated,
    MAX(sr.sale_date) as last_sale_date
FROM products p
LEFT JOIN sales_targets st ON p.id = st.product_id 
    AND st.period_start <= CURDATE() 
    AND st.period_end >= CURDATE()
    AND st.period_type = 'monthly'
LEFT JOIN sales_records sr ON p.id = sr.product_id 
    AND sr.sale_date >= DATE_FORMAT(CURDATE(), '%Y-%m-01')
GROUP BY p.id, st.target_quantity;

-- =====================================================
-- 49. VIEW: products_with_discounts
-- =====================================================
CREATE OR REPLACE VIEW products_with_discounts AS
SELECT 
    p.id,
    p.name,
    p.description,
    p.base_price,
    p.category,
    COALESCE(SUM(ps.quantity), 0) as stock_quantity,
    prd.role_code,
    prd.discount_percent,
    ROUND(p.base_price * (1 - prd.discount_percent / 100), 2) as discounted_price
FROM products p
LEFT JOIN product_stock ps ON p.id = ps.product_id
LEFT JOIN product_role_discounts prd ON p.id = prd.product_id
GROUP BY p.id, prd.role_code, prd.discount_percent;
